# Ansible - Setup Docker Container for Nginx

- name : Setup Docker Container for Nginx
  hosts: all 
  become: yes
  tasks: 
    - name: Pull Nginx Image
      docker_image: 
        name: nginx
        source: pull
    - name: Run Nginx container 
      docker_container: 
        name: nginx-container
        image: nginx
        state: started
        published_ports: 
            - "8080:80"

# Kubernetes - Horizontal Pod Autoscaler ( HPA )

apiVersion: app/v1
kind: Deployment
metadata:
    name: pathnex-Deployment
spec: 
    replicas: 1
    selector: 
        matchLabels:
            app: pathnex-app
    template: 
        metadata: 
            label: 
                app: pathnex-app
        spec: 
            container: 
                - name: nginx
                  image: nginx
                  ports: 
                    - containerPort: 80 

apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata: 
    name: pathnex-hpa
spec: 
    scaleTargetRed: 
        apiVersion: apps/v1
        kind: Deployment
        name: pathnex-Deployment
    miniReplicas: 1
    maxReplicas: 10
    targetCPUUtilizationPercentage: 50
