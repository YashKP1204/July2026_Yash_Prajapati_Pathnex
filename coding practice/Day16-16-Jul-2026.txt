  hosts: localhost
  become: yes
  tasks: 
    - name: Launch EC2 instance in us-east-1
        ec2_instance:
            name: pathnex-ec2-us-east-1
            key_name: pathnex-key 
            instance_type : t2.micro 
            image: ami-0abcd1234abcd12344
            wait: yes
    - name: Launch EC2 instance in us-west-1
        ec2_instance: 
            name: pathnex-ec2-us-west-1
            key_name : pathnex-key
            region: us-west-1
            instance_type: t2.micro 
            image: ami-0abcd1234abcd12344

# Kubernetes - Multi- Cluster Deployment with federation 

apiVersion : app/v1
kind: Deployment 
metaddat: 
    name : pathnex-deployment 
    labels: 
        app: pathnex-app
spec: 
    replicas: 2
    selector: 
        matchLabel: 
            app: pathnex-app
    template: 
        metadata: 
            labels: 
                pathnex-app
        spec: 
            containers: 
                - name: nginx
                  image: nginx
                  ports: 
                    - containerPort: 80

# JenkinsFile = Deploye to Mutli-Cluster Environment 

pipeline {
    agent any 
    stages { 
        stage('Build){
            steps{
                 echo 'Building Docker Image..." 
                 docker.build('pathnex-web-app')
            }
        }
        stage('Test'){
            steps{
                echo 'Running tests..."
            }
        }
        stage('Deploy to Kubernetes'){
            steps {
                script {
                    if (env.CLUSTER == 'us-east'){
                        sh 'kubectl --context=us-east-apply -f kubernetes/deployment.yaml'
                    } else {
                        sh 'kubectl --context-us-west apply -f kubernetes/deployment.yaml'
                    }
                }
            }
        }
    }
}
