# Adcvanced Automation with CloudFormation and CI/CD
# Ansible - Multi-Node Nginx Setup 

- name: Setup Nginx on multiple nodes
  hosts: all 
  become: yes
  tasks: 
    - name: Install Nginx
      yum: 
        name: nginx
        state: present
    - name: Start nginx service
      service: 
        name: nginx
        state: started
        enabled: yes

# Kubernetes - Deploy Nginx wiht ConfigMap and Secret

apiVersion : v1
kind: ConfigMap
metadata: 
    name: nginx-config
data: 
    nginx_conf: |
        server {
            listen 80 ; 
            server_name localhost;
            location / {
                root /usr/share/nginx/html; 
                index index.html index.html;
            }
        }

---

apiVersion: v1
kind: Secret
metadata: 
    name: nginx-secret
type: opaque
data: 
    password: adfertekddkfe2323hdaf=

--- 
apiVersion: apps/v1
kind: Deployment
metadata: 
    name: pathnex-nginx
spec: 
    replicas: 2
    selector: 
        matchLabel: 
            app : pathnex-nginx
    template: 
        metadata: 
           labels 
            app: pathnex-nginx
        spec: 
            containers: 
                - name : nginx
                  image: nginx
                  volumeMounts: 
                    - mountPath: /etc/nginx/nginx.conf
                      name: nginx-config
                      subPath: nginx.conf
            volumes: 
                - name: nginx-config
                  configMap: 
                     name: nginx-config
                - name: nginx-secret
                  secret: 
                    secretName; nginx-secret

# Jenkinsfile- Multi-Branch Pipeline wiht docker Build

pipeline { 
    agent any
    stage {
        stage('Build'){
            steps {
                script {
                    docker.build('pathnex-app')
                }
            }
        }
    }
    stage('Deploy'){
        steps {
            script {
                kubectl apply -f kubernetes/deployment.yaml
            }
        }
    }
}

# Gitlab CI/CD - Advanced Deployment with Helm 

stage: 
    - build
    - push
    - deploy
build:
    stage: build
    script: 
        - docker build -t pathnex-web-app . 
push:
    stage: push 
    script: 
        - docker login -u "$CI_REGISTRY_USER" =p "$CI_REGISTRY_PASSWORD" 
        - docker push pathnex-web-app
deploye: 
    stage: deploy
    script: 
        - helm repo add pathnex https://chart.pathnex.com 
        - helm install pathnex-nginx pathnex/nginx-ingress
