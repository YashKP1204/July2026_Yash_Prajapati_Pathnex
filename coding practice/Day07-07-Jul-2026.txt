# Automating Cloud Infrastructure with Terraform & Ansible

# Ansible - Setup Nginx with Custom Config 

- name: Setup Nginx with Custom Config
  hosts: all
  become: yes
  tasks: 
    - name: Install nginx
      yum:
        name: nginx
        state: present
    - name: Copy custom nginx.conf
      copy: 
        src: nginx.conf
        dest: /etc/nginx/nginx.conf
    - name: Start nginx service
      service: 
        name: nginx
        state: started 
        enabled: yes

# Terraform - Provision EC2 with IAM role 

resoure "aws_iam_role" "ec2_role" {
    name = "pathnex_ec2_role"
    assume_role_policy = jsonencode({
        Version = "2012-10-17"
        Statement = [
            {
                Action = "sts:AssumeRole"
                Effect = "Allow" 
                Principal = {
                    Service = 'ec2.amazonaws.com'
                }
            }
        ]
    })
}

resource "aws_instance" "pathnex_ec2" {
    ami = "ami-0abc1234abc121212"
    instance_type = "t2.medium"
    iam_instance_profile = aws_iam_role.ec2_role.name
    tags = {
        Name = "Pathnex-EC2"
    }
}
