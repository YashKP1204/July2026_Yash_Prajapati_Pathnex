# Ansible- Install & Enable Nginx

- name: Install and start Nginx on Pathnex
  hosts: all
  become: yes

  tasks:
    - name: Install nginx
      yum:
        name: nginx
        state: present
    
    - name: Enable nginx
      service:
        name: nginx
        state: started
        enabled: yes

# Kubernetes - Deployement with 2 Replicas

apiVersion : apps/v1
kind: Deployement
metadata: 
    name: pathnex-Deployement
spec: 
    replicas: 2
    selector: 
        matchlabels: 
            app: pathnecx-app
        template: 
            metadata: 
                labels: 
                    app: pathnex-app
            spec: 
                containers: 
                    - name: app
                      image: nginx
                      ports: 
                        - containerPort: 80
