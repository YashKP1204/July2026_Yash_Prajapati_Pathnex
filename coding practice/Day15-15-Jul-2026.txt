# Automation with Terraform , Ansible , and CI/Cd pipeline

# Ansible - Deploy a Web Application to Multiple hosts

- name: Deploy a web application to multiple hosts
  hosts: all    
  become: yes
  tasks: 
    - name: Copy web app files
      copy: 
        src: /path/to/web/app/
        dest: /var/www/html/
    - name: Start web app service
      service: 
        name: apache2
        state: started
        enabled: yes

# Jenkinsfle - Multi-Environment Deployment 

pipeline {
    agent any 
    environment { 
        IMAGE_NAME = 'pathnex-web-app'
        DEPLOY_ENV = 'production'
    }
    stages{
        stage('Build){
            steps { 
                echo 'Building Dokcer image ...' 
                docker.build("${env.IMAGE_NAME}")
            }
        }
        stage('Test'){
            steps {
                echo 'Runnig tests...'
            }
        }
        stage('Deploye to kubernetes'){
            steps {
                script {
                    if (env.DEPLOY_ENV == 'production'){
                        sh 'kubectl apply -f kubernetes/prod-deployment.yaml'
                    } else {
                        sh 'kubectl apply -f kubernetes/dev-deployment.yaml'
                    }
                }
            }
        }
    }
}

# Gitlab CI/CD - Environ,ment - Specific Deployment 

stages: 
    - build 
    - push 
    - deploy 

build: 
    stage: build 
    script: 
        - docker build -t pathnec-web-app . 

push: 
    stage: push 
    script: 
        - docker login -8u "$CI_REGISTRY_USER" -p "$CI_REGISTRY_PASSWORD" 
        - docker push pathnex-web-app . 

deploy:
    stage: deploy 
    script: 
        - | 
          if [ "$CI_COMMIT_REF_NAME" == "main" ]; then 
            kubectl apply -f kubernetes/prod-deployment.yaml
          else 
            kubectl apply -f kubernetes/prod-deployment.yaml
